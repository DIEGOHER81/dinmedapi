async function generatePdf(entryRequestId, companyCode, options = {}) {
    try {
        log('üöÄ Iniciando generaci√≥n de PDF...', 'info');
        
        // PASO 1: Intentar con DinkToPdf

        const response = await fetch(`/api/Pdf/remision/${entryRequestId}?companyCode=${companyCode}&${new URLSearchParams(options)}`);
        
        if (response.ok) {
            // ‚úì √âxito con DinkToPdf
            log('‚úì PDF generado exitosamente con DinkToPdf', 'success');
            const blob = await response.blob();
            downloadPdf(blob, `Remision_P-${entryRequestId}.pdf`);
            return { success: true, method: 'DinkToPdf' };
        }
        
        if (response.status === 503) {
            // PASO 2: Fallback a HTML
            log('PASO 2: DinkToPdf fall√≥, usando fallback HTML...', 'step-2');
            const errorData = await response.json();
            
            if (errorData.fallback && errorData.fallback.url) {
                log('‚úì Enlace HTML disponible', 'success');
                openHtmlInNewWindow(errorData.fallback.url);
                return { success: true, method: 'HTML', url: errorData.fallback.url };
            }
        }
        
        throw new Error(`Error ${response.status}: ${response.statusText}`);
        
    } catch (error) {
        log(`‚úó Error: ${error.message}`, 'error');
        return { success: false, error: error.message };
    }
}
